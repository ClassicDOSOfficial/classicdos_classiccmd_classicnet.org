ClassicCMD is the emissary shell of the Classic Series: a batch dispatcher (classiccmd.bat) routing verbs to a modular Python command suite in \COMMANDS. It is approved as the official interface twin to ClassicDOS.

Root layout

ClassicCMD\
  classiccmd.bat
  INSTALL.bat
  CORE\
    env.bat, banner.bat, version.bat, help.bat, alias.bat
  COMMANDS\
    20 Python commands (*.py)
Dispatcher excerpt

@echo off
setlocal ENABLEDELAYEDEXPANSION
call "%~dp0CORE\env.bat"
call "%~dp0CORE\banner.bat"
:loop
set /p "CMDLINE=%USERNAME%@ClassicCMD> "
for /f "tokens=1,* delims= " %%A in ("!CMDLINE!") do (
  set "VERB=%%A" & set "ARGS=%%B"
)
if exist "%~dp0COMMANDS\!VERB!.py" (
  python "%~dp0COMMANDS\!VERB!.py" !ARGS!
) else (
  echo Unknown command: !VERB!
)
goto loop
